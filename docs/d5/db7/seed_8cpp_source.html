133  void Seed::applyPatches( bool set )
134  {
135  using namespace libtp;
136 
137  m_SeedStatus |= SEED_STATUS_PATCHED;
138 
139  uint32_t num_patches = m_Header->patchInfo.numEntries;
140  uint32_t gci_offset = m_Header->patchInfo.dataOffset;
141 
142  // Don't bother to patch anything if there's nothing to patch
143  if ( num_patches > 0 )
144  {
145  uint32_t num_bytes = num_patches / 8 + 1;
146 
147  // Set the pointer as offset into our buffer
148  uint8_t* patch_config = &m_GCIData[gci_offset];
149 
150  for ( uint32_t i = 0; i < num_bytes; i++ )
151  {
152  uint8_t byte = patch_config[i];
153 
154  for ( uint32_t b = 0; b < 8; b++ )
155  {
156  if ( ( byte >> b ) & 0x80 )
157  {
158  // run the patch function for this bit index
159  uint32_t index = i * 8 + b;
160 
161  using namespace game_patch;
162  if ( index < sizeof( patches ) / sizeof( patches[0] ) )
163  {
164  patches[index]( set );
165  m_PatchesApplied++;
166  }
167  }
168  }
169  }

200  void Seed::applyRegionFlags()
201  {
202  using namespace libtp;
203 
204  m_SeedStatus |= SEED_STATUS_REGIONFLG;
205 
206  uint32_t num_regionFlags = m_Header->regionFlagsInfo.numEntries;
207  uint32_t gci_offset = m_Header->regionFlagsInfo.dataOffset;
208 
209  if ( num_regionFlags > 0 )
210  {
211  tp::d_com_inf_game::GameInfo* gameInfo = &tp::d_com_inf_game::dComIfG_gameInfo;
212 
213  // Set the pointer as offset into our buffer
214  regionFlag* regionFlags = reinterpret_cast<regionFlag*>( &m_GCIData[gci_offset] );
215 
216  uint8_t regionID = data::stage::regions[0];
217  // Loop through all regions
218  for ( uint32_t i = 0; i < sizeof( data::stage::regions ); i++ )
219  {
220  regionID = data::stage::regions[i];
221 
222  tp::d_save::getSave( gameInfo, regionID );
223 
224  // Loop through region-flags from gci
225  for ( uint32_t j = 0; j < num_regionFlags; j++ )
226  {
227  if ( regionFlags[j].region_id == regionID )
228  {
229  // The n'th bit that we want to set TRUE
230  int32_t bit = regionFlags[j].bit_id;
231 
232  int32_t offset = bit / 8;
233  int32_t shift = bit % 8;
234 
235  // Failsafe; localAreaNode size is 0x20
236  if ( offset < 0x20 )
237  {
238  gameInfo->localAreaNodes[offset] |= ( 0x80 >> shift );
239  m_AreaFlagsModified++;
240  }
241  }
242  }
243 
244  // Save the modifications
245  tp::d_save::putSave( gameInfo, regionID );
246  }
247 
248  // Reset to previous region ID
249  uint8_t i = 0;
250 
251  while ( !strcmp( data::stage::allStages[i], gameInfo->currentStage ) )
252  {
253  i++;
254  }
255 
256  tp::d_save::getSave( gameInfo, data::stage::regions[i] );
257  }
