73  libtp::tp::dzx::ACTR* dzxData = reinterpret_cast<libtp::tp::dzx::ACTR*>( chunkTypeInfo->chunkDataPtr );
74 
75  // Check if we have DZX checks to work with
76  if ( !numReplacements )
77  return;
78 
79  // Loop through all chunks the game is currently loading/setting
80  for ( uint32_t i = 0; i < numChunks; i++ )
81  {
82  // The hash in RAM right now
83  uint16_t actorHash =
84  libtp::tools::fletcher16( reinterpret_cast<uint8_t*>( &dzxData[i] ), sizeof( libtp::tp::dzx::ACTR ) );
85 
86  // Compare to all available replacements
87  for ( uint32_t j = 0; j < numReplacements; j++ )
88  {
89  if ( dzxReplacements[j].hash == actorHash )
90  {
91  // Bytearray of target ACTR struct
92  uint8_t* target = reinterpret_cast<uint8_t*>( &dzxData[i] );
93 
94  // Replace target Actor with replacement values if != FF
95  for ( uint8_t b = 0; b < sizeof( libtp::tp::dzx::ACTR ); b++ )
96  {
97  // Fetch replacement byte
